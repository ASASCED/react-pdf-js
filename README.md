## Install

```bash
npm install --save react-pdf-js
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-pdf-js'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
